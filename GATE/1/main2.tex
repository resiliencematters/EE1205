\end{align}
Using Contour Integration to find the inverse $Z$-transform,
\begin{align}
    y(49)&=\frac{1}{2\pi j}\oint_{C}Y(z) \;z^{48} \;dz  \\
    &=\frac{1}{2\pi j}\oint_{C}\frac{(z^{-1}+1)z^{48}}{({1-z^{-1})}^{3}} \;dz 
\end{align}
We can observe that the pole is repeated $3$ times and thus $m=3$,
\begin{align}
    R&=\frac{1}{\brak {m-1}!}\lim\limits_{z\to a}\frac{d^{m-1}}{dz^{m-1}}\brak {{(z-a)}^{m}f\brak z}  \\
    \implies R &=\frac{1}{\brak {2}!}\lim\limits_{z\to 1}\frac{d^{2}}{dz^{2}}\brak {{(z-1)}^{3}\frac{(z^{-1}+1)z^{51}}{{(z-1)}^3}}   \\
    \implies R &=\frac{1}{2}\lim\limits_{z\to 1}\frac{d^2}{dz^2}(z^{50}+z^{51})   \\
    \implies R &=2500 \\
    \therefore y(50)&=2500
